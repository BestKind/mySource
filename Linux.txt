配置虚拟主机站点
server {
        listen       80;
        server_name s02.51vx.cn;
        index index.html index.htm index.php;
        root  /home/website/activity/public;

        location / {
                if (!-e $request_filename) {
                        rewrite . /index.php last;
                        break;
                }
        }

        location ~ .*\.(php|php5)?$ {
                fastcgi_pass 127.0.0.1:9000;
                fastcgi_index index.php;
                fastcgi_param  SCRIPT_FILENAME /home/website/activity/public$fastcgi_script_name;
                include fastcgi_params;
        }

        access_log  /home/website/activity/log/activity.log;
}
